What This Does

A dual-joystick USB game controller built on Arduino, communicating over serial to a Python script that translates physical inputs into keyboard and mouse events for FreeFire on Windows. No drivers needed — pure ctypes + pynput automation.

ANALOG INPUTS

DIGITAL INPUTS (INPUT_PULLUP)

Arduino PinFunctionAction
D2LSW — Left Joy ClickScope / ADS
D3RSW — Right Joy ClickFire (LMB)
D4BTN1Spacebar (Jump)
D5BTN2G (Grenade)
D6BTN3Q (Ability)
D7BTN4R (Reload)
⚠ INPUT_PULLUP NOTE
All digital pins use INPUT_PULLUP. This means 0 = pressed, 1 = released. The Python script accounts for this with the lsw == 0 and btn == 0 checks.
// 04 — arduino firmware

Arduino Code — arduino_v4.ino

C++ / Arduino arduino_v4.ino 
const int LX_PIN  = A0;  // Left Joy X  → Mouse horizontal
const int LY_PIN  = A1;  // Left Joy Y  → Mouse vertical
const int LSW_PIN = 2;   // Left Joy Click → Scope (Right Mouse)
const int RX_PIN  = A2;  // Right Joy X → A / D keys
const int RY_PIN  = A3;  // Right Joy Y → W / S keys
const int RSW_PIN = 3;   // Right Joy Click → Fire (Left Mouse)
const int BTN1_PIN = 4;  // Spacebar
const int BTN2_PIN = 5;  // G key
const int BTN3_PIN = 6;  // Q key
const int BTN4_PIN = 7;  // R key

void setup() {
  Serial.begin(9600);
  pinMode(LSW_PIN,  INPUT_PULLUP);  // 0=pressed, 1=released
  pinMode(RSW_PIN,  INPUT_PULLUP);  // 0=pressed, 1=released
  pinMode(BTN1_PIN, INPUT_PULLUP);
  pinMode(BTN2_PIN, INPUT_PULLUP);
  pinMode(BTN3_PIN, INPUT_PULLUP);
  pinMode(BTN4_PIN, INPUT_PULLUP);
}

void loop() {
  int lx  = analogRead(LX_PIN);   // 0–1023
  int ly  = analogRead(LY_PIN);   // 0–1023
  int lsw = digitalRead(LSW_PIN); // 0=clicked
  int rx  = analogRead(RX_PIN);   // 0–1023
  int ry  = analogRead(RY_PIN);   // 0–1023
  int rsw = digitalRead(RSW_PIN); // 0=clicked
  int btn1 = digitalRead(BTN1_PIN);
  int btn2 = digitalRead(BTN2_PIN);
  int btn3 = digitalRead(BTN3_PIN);
  int btn4 = digitalRead(BTN4_PIN);

  Serial.print(lx);   Serial.print(',');
  Serial.print(ly);   Serial.print(',');
  Serial.print(lsw);  Serial.print(',');
  Serial.print(rx);   Serial.print(',');
  Serial.print(ry);   Serial.print(',');
  Serial.print(rsw);  Serial.print(',');
  Serial.print(btn1); Serial.print(',');
  Serial.print(btn2); Serial.print(',');
  Serial.print(btn3); Serial.print(',');
  Serial.println(btn4);

  delay(20);
}

Python Script — final.py

Python 3 final.py — Imports & Config 
# pip install pyserial keyboard pynput
# Run as ADMINISTRATOR

import serial
import time
import sys
import keyboard
import ctypes
from pynput.mouse import Button, Controller as MouseController

# ── Config ────────────────────────────────────────
COM_PORT        = 'COM3'
BAUD_RATE       = 9600
DEADZONE_LOW    = 400
DEADZONE_HIGH   = 624
AIM_SENSITIVITY = 20

# ── Low-level mouse MOVE only (for in-game camera) ─
class MOUSEINPUT(ctypes.Structure):
    _fields_ = [
        ("dx",          ctypes.c_long),
        ("dy",          ctypes.c_long),
        ("mouseData",   ctypes.c_ulong),
        ("dwFlags",     ctypes.c_ulong),
        ("time",        ctypes.c_ulong),
        ("dwExtraInfo", ctypes.POINTER(ctypes.c_ulong)),
    ]

class INPUT(ctypes.Structure):
    class _INPUT(ctypes.Union):
        _fields_ = [("mi", MOUSEINPUT)]
    _anonymous_ = ("_input",)
    _fields_ = [("type", ctypes.c_ulong), ("_input", _INPUT)]

def send_mouse_move(dx, dy):
    inp = INPUT()
    inp.type = 0
    inp.mi.dx = dx
    inp.mi.dy = dy
    inp.mi.mouseData = 0
    inp.mi.dwFlags = 0x0001  # MOUSEEVENTF_MOVE
    inp.mi.time = 0
    inp.mi.dwExtraInfo = None
    ctypes.windll.user32.SendInput(1, ctypes.byref(inp), ctypes.sizeof(inp))

# ── Mouse (pynput) for clicks only ────────────────
mc         = MouseController()
fire_held  = False
scope_held = False
w_held = a_held = s_held = d_held = False
space_held = False
g_held     = False
q_held     = False
r_held     = False

def delta(v):
    if DEADZONE_LOW < v < DEADZONE_HIGH:
        return 0
    n = (v - 512) / 512.0
    return int(n * abs(n) * AIM_SENSITIVITY)
Python 3 final.py — Connect & Main Loop 
# ── Connect ───────────────────────────────────────
print(f"Connecting to {COM_PORT}...")
try:
    ser = serial.Serial(COM_PORT, BAUD_RATE, timeout=1)
except Exception as e:
    print(f"ERROR: {e}"); sys.exit(1)
time.sleep(2)
ser.flushInput()
print("Ready! Push RIGHT joystick for WASD, LEFT joystick for mouse.")
print("Ctrl+C to stop.\n")

# ── Main loop ─────────────────────────────────────
try:
    while True:
        try:
            line = ser.readline().decode('utf-8').strip()
            if not line: continue
            p = line.split(',')
            if len(p) != 10: continue
            lx, ly, lsw, rx, ry, rsw, btn1, btn2, btn3, btn4 = [int(x) for x in p]
        except: continue

        # RIGHT joystick → WASD (axes swapped)
        if ry > DEADZONE_HIGH and not w_held:
            keyboard.press('w'); w_held = True
        elif ry <= DEADZONE_HIGH and w_held:
            keyboard.release('w'); w_held = False

        if ry < DEADZONE_LOW and not s_held:
            keyboard.press('s'); s_held = True
        elif ry >= DEADZONE_LOW and s_held:
            keyboard.release('s'); s_held = False

        if rx > DEADZONE_HIGH and not a_held:
            keyboard.press('a'); a_held = True
        elif rx <= DEADZONE_HIGH and a_held:
            keyboard.release('a'); a_held = False

        if rx < DEADZONE_LOW and not d_held:
            keyboard.press('d'); d_held = True
        elif rx >= DEADZONE_LOW and d_held:
            keyboard.release('d'); d_held = False

        # BTN1 → Spacebar
        if btn1 == 0 and not space_held:
            keyboard.press('space'); space_held = True
        elif btn1 == 1 and space_held:
            keyboard.release('space'); space_held = False

        # BTN2 → G key
        if btn2 == 0 and not g_held:
            keyboard.press('g'); g_held = True
        elif btn2 == 1 and g_held:
            keyboard.release('g'); g_held = False

        # BTN3 → Q key
        if btn3 == 0 and not q_held:
            keyboard.press('q'); q_held = True
        elif btn3 == 1 and q_held:
            keyboard.release('q'); q_held = False

        # BTN4 → R key
        if btn4 == 0 and not r_held:
            keyboard.press('r'); r_held = True
        elif btn4 == 1 and r_held:
            keyboard.release('r'); r_held = False

        # RIGHT click → Scope (RSW)
        if rsw == 0 and not scope_held:
            mc.press(Button.right); scope_held = True
        elif rsw == 1 and scope_held:
            mc.release(Button.right); scope_held = False

        # LEFT joystick → Camera/Head aim
        mx, my = delta(lx), delta(ly)
        if mx or my: send_mouse_move(mx, my)

        # LEFT click → Fire (LSW)
        if lsw == 0 and not fire_held:
            mc.press(Button.left); fire_held = True
        elif lsw == 1 and fire_held:
            mc.release(Button.left); fire_held = False

except KeyboardInterrupt:
    if w_held:     keyboard.release('w')
    if a_held:     keyboard.release('a')
    if s_held:     keyboard.release('s')
    if d_held:     keyboard.release('d')
    if space_held: keyboard.release('space')
    if g_held:     keyboard.release('g')
    if q_held:     keyboard.release('q')
    if r_held:     keyboard.release('r')
    if fire_held:  mc.release(Button.left)
    if scope_held: mc.release(Button.right)
    ser.close()
    print("Stopped.")
// 06 — control logic

Input → Action Mapping

🕹 LEFT JOYSTICK — AIM
X > 624→Mouse RIGHT
X < 400→Mouse LEFT
Y > 624→Mouse DOWN
Y < 400→Mouse UP
400–624→Deadzone (no move)
Click (LSW=0)→Left Click (Fire)
🕹 RIGHT JOYSTICK — MOVE
X > 624→Press A
X < 400→Press D
Y > 624→Press W
Y < 400→Press S
400–624→Release all WASD
Click (RSW=0)→Right Click (Scope)
🔵 PUSH BUTTONS
BTN1 (D4=0)→Space (Jump)
BTN2 (D5=0)→G (Grenade)
BTN3 (D6=0)→Q (Ability)
BTN4 (D7=0)→R (Reload)
Any BTN=1→Released
📐 SENSITIVITY FORMULA
d = (v − 512) / 512.0
normalize to −1.0 … +1.0
move = int(n × |n| × sensitivity)
quadratic scaling for fine aim
sensitivity = 20 (adjustable)

STEPS TO RUN

01 Upload arduino_v4.ino to Arduino via IDE. Check COM3 matches your port.
02 Open terminal as Administrator. Navigate to project folder.
03 Run: python final.py — you'll see "Ready! Push RIGHT joystick…"
04 Launch FreeFire. Controller is live. Ctrl+C to stop cleanly.
⚠ TUNING TIP
Change DEADZONE_HIGH, DEADZONE_LOW, and AIM_SENSITIVITY at the top of final.py to tune feel. The default deadzone is 400–624 (out of 0–1023) to compensate for joystick drift at center. Default sensitivity is 20.
